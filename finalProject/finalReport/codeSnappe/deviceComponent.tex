        const deviceComponent = () => {
            const {
                data: deviceData, 
                loading: deviceLoading, 
                error: deviceError,
                getDeviceList
                } = useGetDevicelist();
            
            useEffect(() => {
                getDeviceList();
            }, []);

            return (
                <>
                    {deviceError && 
                        <div>Failed to load device list</div>}
                    {deviceLoading && 
                        <div>Loading...</div>}
                    {!deviceLoading && !deviceError && 
                        <TableComponent data={deviceData} />}
                </>
            )
        }
